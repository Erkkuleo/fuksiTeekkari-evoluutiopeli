(Säännöt ison pelaajan liikuttamiseen)

[ > Player | Body ] -> [ > Player | > Body ]
[ ^ Player | Body ] -> [ ^ Player | ^ Body ]
[ V Player | Body ] -> [ V Player | V Body ]
[ < Player | Body ] -> [ < Player | < Body ]
[ > Body | Body ] -> [ > Body | > Body ]
[ < Body | Body ] -> [ < Body | < Body ]
[ V Body | Body ] -> [ V Body | V Body ]
[ ^ Body | Body ] -> [ ^ Body | ^ Body ]
[ > Body | Wall ] -> cancel

[> Player ] [ muuriPelaaja ] -> [> Player ][> muuriPelaaja]


(NPC random liikkuminen)

[stationary npc1] -> [randomDir npc1]

(npc liikkuminen)

[ > npc | Body ] -> [ > npc | > Body ]
[ ^ npc | Body ] -> [ ^ npc | ^ Body ]
[ V npc | Body ] -> [ V npc | V Body ]
[ < npc | Body ] -> [ < npc | < Body ]
[ > Body | Body ] -> [ > Body | > Body ]
[ < Body | Body ] -> [ < Body | < Body ]
[ V Body | Body ] -> [ V Body | V Body ]
[ ^ Body | Body ] -> [ ^ Body | ^ Body ]
[ > Body | Wall ] -> cancel
(npc suojamuurin liikuttaminen)

([> npc ] [muuriNpc] -> [> npc ][> muuriNpc])


(Tappelumoden rulet)
horizontal [> Player ] [ choosebox ] -> [> Player ][> choosebox]

[ Action player] [rock choosebox ] [ chosen | enemyChosen ] -> [player ][rock] [ rock | enemyChoosing]
[ Action player] [paper choosebox ] [ chosen | enemyChosen] -> [player ][paper] [ paper | enemyChoosing]
[ Action player] [scissors choosebox ] [ chosen | enemyChosen] -> [player ][scissors] [ scissors | enemyChoosing]

[enemyChoosing] -> [random options]

[muuriPelaaja | muuriNpc] -> win
